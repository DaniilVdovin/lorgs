{# Macro to create the timeline ruler part, which is usually shown at the top or bottom of a timeline. #} {% macro timeline_ruler(duration, show_killtime=False) -%}
{% for i in range(0, duration, 10*1000) -%}
{%- if i % 30000 == 0 -%}
{{ i|format_time }}
{%- endif -%}
{% endfor %}
{# show killtime (or wipetime) #} {% if show_killtime %}
{{duration|format_time}}
{% endif %}
{%- endmacro %}
